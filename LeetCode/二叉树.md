# 二叉树总结
## 二叉树的最大深度
方法：**二叉树的中序遍历** + **递归**  
 c++代码
```c++
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int maxDepth(TreeNode* root) {
        if(!root)
            return 0;
        
        int num=1;
        //if(root->left)
        int num_l=maxDepth(root->left);
        int num_r=maxDepth(root->right);
        num+=num_l>num_r?num_l:num_r;
        return num;
    }
};
```
